Usage Snippet 
import Pulsar from "./src/index.ts";

const pulsar = new Pulsar();
const composer = pulsar.getPatternComposer();

composer.parse([
  {
    type: "line",
    timestamp: 0,
    duration: 540,
    intensity: [
      { time: 0, value: 0.16 },
      { time: 180, value: 0.32 },
      { time: 360, value: 0.68 },
      { time: 540, value: 0.96 },
    ],
    frequency: [
      { time: 0, value: 0.2 },
      { time: 220, value: 0.42 },
      { time: 420, value: 0.74 },
      { time: 540, value: 0.9 },
    ],
  },
]);

composer.play();
